__풀이__  
```python
pattern = [ "WBWBWBWB",
            "BWBWBWBW"]

def check(board, y, x):
    ret0 = 0
    ret1 = 0
    for i in range(8):
        for j in range(8):
            if board[y+i][x+j] != pattern[i%2][j]:
                ret0 += 1

    for i in range(8):
        for j in range(8):
            if board[y+i][x+j] != pattern[(i+1)%2][j]:
                ret1 += 1

    return min(ret0, ret1)

N, M = map(int, input().split())
board = [ input() for _ in range(N) ]
result = 32

for i in range(N-7):
    for j in range(M-7):
        num = check(board, i, j)
        if result > num:
            result = num
print(result)
```
